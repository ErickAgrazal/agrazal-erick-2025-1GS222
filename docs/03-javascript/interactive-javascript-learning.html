📚 Fundamentos de JavaScript

💡 ¿Qué es JavaScript?

JavaScript es un lenguaje de programación que permite crear sitios web interactivos. Es el único lenguaje que los navegadores pueden ejecutar nativamente.

📦 Variables y Tipos de Datos

🎯 Variables

Las variables son contenedores para almacenar valores. JavaScript tiene tres formas de declarar variables: var, let y const.

🔀 Control de Flujo

🎛️ Estructuras de Control

Las estructuras de control permiten tomar decisiones y repetir acciones en tu código.

control.js
// IF - ELSE console.log("=== IF - ELSE ==="); let edad = 20; if (edad >= 18) { console.log("Eres mayor de edad"); } else { console.log("Eres menor de edad"); } // IF - ELSE IF - ELSE let hora = new Date().getHours(); console.log(`\nSon las ${hora} horas`); if (hora < 12) { console.log("Buenos días"); } else if (hora < 18) { console.log("Buenas tardes"); } else { console.log("Buenas noches"); } // SWITCH console.log("\n=== SWITCH ==="); let dia = new Date().getDay(); let nombreDia; switch (dia) { case 0: nombreDia = "Domingo"; break; case 1: nombreDia = "Lunes"; break; case 2: nombreDia = "Martes"; break; case 3: nombreDia = "Miércoles"; break; case 4: nombreDia = "Jueves"; break; case 5: nombreDia = "Viernes"; break; case 6: nombreDia = "Sábado"; break; default: nombreDia = "Día inválido"; } console.log(`Hoy es ${nombreDia}`); // FOR LOOP console.log("\n=== FOR LOOP ==="); console.log("Contando del 1 al 5:"); for (let i = 1; i <= 5; i++) { console.log(`Iteración ${i}`); } console.log("\nTabla del 3:"); for (let i = 1; i <= 10; i++) { console.log(`3 x ${i} = ${3 * i}`); } // WHILE LOOP console.log("\n=== WHILE LOOP ==="); let contador = 0; console.log("Contando con while:"); while (contador < 5) { console.log(`Contador: ${contador}`); contador++; } // DO-WHILE LOOP console.log("\n=== DO-WHILE LOOP ==="); let numero = 1; console.log("Números pares hasta 10:"); do { if (numero % 2 === 0) { console.log(numero); } numero++; } while (numero <= 10); // FOR...OF (para arrays) console.log("\n=== FOR...OF ==="); let colores = ["Rojo", "Verde", "Azul"]; console.log("Iterando sobre colores:"); for (let color of colores) { console.log(`Color: ${color}`); } // FOR...IN (para objetos) console.log("\n=== FOR...IN ==="); let persona = { nombre: "Carlos", edad: 25, ciudad: "Panamá" }; console.log("Propiedades de persona:"); for (let propiedad in persona) { console.log(`${propiedad}: ${persona[propiedad]}`); } // BREAK Y CONTINUE console.log("\n=== BREAK Y CONTINUE ==="); console.log("Números del 1 al 10 (saltando el 5, parando en 8):"); for (let i = 1; i <= 10; i++) { if (i === 5) { continue; // Saltar esta iteración } if (i === 8) { break; // Salir del loop } console.log(i); } // EJEMPLO PRÁCTICO: Adivina el número console.log("\n=== JUEGO: ADIVINA EL NÚMERO ==="); let numeroSecreto = Math.floor(Math.random() * 10) + 1; let intentos = 3; console.log("Adivina un número del 1 al 10. Tienes 3 intentos."); // Simulamos intentos let intentosUsuario = [5, 7, numeroSecreto]; // El último es el correcto for (let i = 0; i < intentos; i++) { let intento = intentosUsuario[i]; console.log(`Intento ${i + 1}: ${intento}`); if (intento === numeroSecreto) { console.log("¡Felicidades! ¡Adivinaste!"); break; } else if (intento < numeroSecreto) { console.log("El número es mayor"); } else { console.log("El número es menor"); } if (i === intentos - 1 && intento !== numeroSecreto) { console.log(`Game Over. El número era ${numeroSecreto}`); } }
